001package edu.neu.csye6200.manager;
002
003import edu.neu.csye6200.helper.Log;
004import edu.neu.csye6200.helper.SQLUtils;
005import edu.neu.csye6200.model.*;
006import org.jdbi.v3.core.Jdbi;
007import org.jdbi.v3.core.mapper.reflect.ReflectionMappers;
008import org.jdbi.v3.sqlite3.SQLitePlugin;
009import org.jdbi.v3.sqlobject.SqlObjectPlugin;
010
011import java.sql.Connection;
012import java.sql.DriverManager;
013import java.sql.SQLException;
014
015public enum DatabaseManager {
016        INSTANCE;
017        public static DatabaseManager getInstance()
018        {
019                return INSTANCE;
020        }
021        public static Jdbi getDB()
022        {
023                return DatabaseManager.getInstance().getDataBase();
024        }
025
026        DatabaseManager()
027        {
028        }
029
030        private Connection connection;
031        public boolean isConnectionWorking()
032        {
033                try {
034                        return connection != null && !connection.isClosed();
035                } catch (SQLException throwable) {
036                        throwable.printStackTrace();
037                }
038                return false;
039        }
040        private Jdbi jdbi;
041        public Jdbi getDataBase()
042        {
043                return jdbi;
044        }
045
046        public void use(String path)
047        {
048                Log.i("DatabaseManager.use(" + path + ")");
049                try {
050                        connection = DriverManager.getConnection("jdbc:sqlite:"+path);
051                        jdbi = Jdbi.create(connection).installPlugin(new SqlObjectPlugin()).installPlugin(new SQLitePlugin());
052                        jdbi.getConfig(ReflectionMappers.class).setStrictMatching(false);
053
054                        jdbi.onDemand(TeacherDao.class).createTable();
055                        jdbi.onDemand(StudentDao.class).createTable(SQLUtils.getTableName(Student.class), SQLUtils.getProperties(Student.class));
056                        jdbi.onDemand(ParentDao.class).createTable(SQLUtils.getTableName(Parent.class), SQLUtils.getProperties(Parent.class));
057
058                } catch (SQLException throwable) {
059                        Log.e(throwable.getMessage());
060                        throwable.printStackTrace();
061                }
062                Log.i("DatabaseManager connection " + (connection != null ? "established" : "lost"));
063        }
064}
