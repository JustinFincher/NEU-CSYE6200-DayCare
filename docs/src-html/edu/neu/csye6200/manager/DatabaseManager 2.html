001package edu.neu.csye6200.manager;
002
003import edu.neu.csye6200.helper.Log;
004import edu.neu.csye6200.helper.SQLUtils;
005import edu.neu.csye6200.model.*;
006import org.jdbi.v3.core.Jdbi;
007import org.jdbi.v3.core.mapper.reflect.ReflectionMappers;
008import org.jdbi.v3.sqlite3.SQLitePlugin;
009import org.jdbi.v3.sqlobject.SqlObjectPlugin;
010
011import java.io.File;
012import java.sql.Connection;
013import java.sql.DriverManager;
014import java.sql.SQLException;
015
016public enum DatabaseManager {
017        INSTANCE;
018        public static DatabaseManager getInstance()
019        {
020                return INSTANCE;
021        }
022        public static Jdbi getDB()
023        {
024                return DatabaseManager.getInstance().getDataBase();
025        }
026
027        DatabaseManager()
028        {
029        }
030
031        public File getFile() { return new File(DatabaseManager.getInstance().getPath()); }
032        public String getPath() { return path; }
033        private String path;
034        private Connection connection;
035        public boolean isConnectionWorking()
036        {
037                try {
038                        return connection != null && !connection.isClosed();
039                } catch (SQLException throwable) {
040                        throwable.printStackTrace();
041                }
042                return false;
043        }
044        private Jdbi jdbi;
045        public Jdbi getDataBase()
046        {
047                return jdbi;
048        }
049
050        public void use(String path)
051        {
052                this.path = path;
053                Log.i("DatabaseManager.use(" + path + ")");
054                try {
055                        connection = DriverManager.getConnection("jdbc:sqlite:"+path);
056                        jdbi = Jdbi.create(connection).installPlugin(new SqlObjectPlugin()).installPlugin(new SQLitePlugin());
057                        jdbi.getConfig(ReflectionMappers.class).setStrictMatching(false);
058                        jdbi.onDemand(TeacherDao.class).createTable();
059                        jdbi.onDemand(StudentDao.class).createTable(SQLUtils.getTableName(Student.class), SQLUtils.getProperties(Student.class));
060                        jdbi.onDemand(ParentDao.class).createTable(SQLUtils.getTableName(Parent.class), SQLUtils.getProperties(Parent.class));
061                        jdbi.onDemand(RatioRuleDao.class).createTable(SQLUtils.getTableName(RatioRule.class), SQLUtils.getProperties(RatioRule.class));
062                } catch (SQLException throwable) {
063                        Log.e(throwable.getMessage());
064                        throwable.printStackTrace();
065                }
066                Log.i("DatabaseManager connection " + (connection != null ? "established" : "lost"));
067        }
068}
