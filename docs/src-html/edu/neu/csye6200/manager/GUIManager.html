009public enum GUIManager {
010        INSTANCE;
011        public static GUIManager getInstance()
012        {
013                return INSTANCE;
014        }
015
016        GUIManager()
017        {
018                if (System.getProperty("os.name", "").startsWith("Mac OS")) {
019                        System.setProperty("apple.laf.useScreenMenuBar", "true");
020                        System.setProperty("apple.awt.textantialiasing", "true");
021                }
022
023                FlatIntelliJLaf.install();
024        }
025
026        public void start()
027        {
028                new ApplicationFormController();
029        }
