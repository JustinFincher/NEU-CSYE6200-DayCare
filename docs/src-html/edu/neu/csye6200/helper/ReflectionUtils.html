001package edu.neu.csye6200.helper;
002
003import javax.annotation.Nonnull;
004import java.lang.reflect.Field;
005import java.util.ArrayList;
006import java.util.Arrays;
007import java.util.List;
008
009public class ReflectionUtils
010{
011    public static List<Field> getAllFields(@Nonnull List<Field> fields, Class<?> type) {
012        fields.addAll(Arrays.asList(type.getDeclaredFields()));
013        if (type.getSuperclass() != null) {
014            getAllFields(fields, type.getSuperclass());
015        }
016        return fields;
017    }
018    public static List<Field> getAllFields(Class<?> type) {
019        return getAllFields(new ArrayList<Field>(), type);
020    }
021}
