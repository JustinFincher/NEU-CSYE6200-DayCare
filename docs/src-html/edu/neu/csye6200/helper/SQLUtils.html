001package edu.neu.csye6200.helper;
002
003import com.google.common.base.CaseFormat;
004import com.google.common.base.Function;
005import edu.neu.csye6200.model.DBObject;
006
007import java.beans.PropertyDescriptor;
008import java.lang.reflect.InvocationTargetException;
009import java.util.HashMap;
010import java.util.Iterator;
011import java.util.List;
012import java.util.Map;
013import java.util.stream.Collectors;
014import java.util.stream.Stream;
015
016public class SQLUtils
017{
018    public static String getKeyInString(PropertyDescriptor descriptor)
019    {
020        return CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_UNDERSCORE, descriptor.getName());
021    }
022
023    public static String getAllKeysInString(Class<? extends DBObject> cls)
024    {
025        StringBuilder sb = new StringBuilder();
026        Iterator<PropertyDescriptor> iterator = BeanUtils.getBeanProperties(cls).iterator();
027        while (iterator.hasNext()) {
028            PropertyDescriptor propertyDescriptor = iterator.next();
029            String dbKey = getKeyInString(propertyDescriptor);
030            Class<?> returnType = propertyDescriptor.getReadMethod().getReturnType();
031            String sql = getType(dbKey, returnType);
032            if (!sql.isEmpty())
033            {
034                sb.append(sql);
035                if (iterator.hasNext()) {
036                    sb.append(", ");
037                }
038            }
039        }
040        return sb.toString();
041    }
042
043    public static List<String> getAllKeysInList(Class<? extends DBObject> cls)
044    {
045        return BeanUtils.getBeanProperties(cls).stream().map(SQLUtils::getKeyInString).collect(Collectors.toList());
046    }
047
048    public static <T extends DBObject> Map<String, ?> getKeysAndValuesInMap(T obj)
049    {
050        return BeanUtils.getBeanProperties(obj.getClass()).stream()
051                .collect(HashMap::new, (m,v) -> {
052                    try {
053                        m.put(SQLUtils.getKeyInString(v), v.getReadMethod().invoke(obj));
054                    } catch (IllegalAccessException | InvocationTargetException e) {
055                        e.printStackTrace();
056                    }
057                }, HashMap::putAll);
058    }
059
060    public static <T extends DBObject> String getKeysAndValueBindingsInCreateString(T obj, boolean includeId)
061    {
062        StringBuilder keys = new StringBuilder();
063        StringBuilder values = new StringBuilder();
064        Iterator<PropertyDescriptor> iterator = BeanUtils.getBeanProperties(obj.getClass()).iterator();
065        keys.append("(");
066        values.append("VALUES(");
067        while (iterator.hasNext()) {
068            PropertyDescriptor propertyDescriptor = iterator.next();
069            String name = propertyDescriptor.getName();
070            String dbKey = CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_UNDERSCORE, name);
071            if (!dbKey.equals("id") || includeId)
072            {
073                keys.append(dbKey);
074                values.append(":").append(dbKey);
075                if (iterator.hasNext()) {
076                    keys.append(", ");
077                    values.append(", ");
078                }
079            }
080        }
081        keys.append(")");
082        values.append(")");
083        return keys.append(" ").append(values).toString();
084    }
085
086    public static <T extends DBObject> String getKeysAndValueBindingsInUpdateString(T obj, boolean includeId)
087    {
088        Stream<String> stream = getAllKeysInList(obj.getClass())
089                .stream();
090        if (!includeId)
091        {
092            stream = stream.filter(s -> !s.equals("id"));
093        }
094        return stream
095                .map((Function<String, String>) input -> input + " = :" + input)
096                .collect(Collectors.joining(", "));
097    }
098
099    public static <T extends DBObject> String getKeysAndValues(T obj, boolean includeId)
100    {
101        StringBuilder keys = new StringBuilder();
102        StringBuilder values = new StringBuilder();
103        Iterator<PropertyDescriptor> iterator = BeanUtils.getBeanProperties(obj.getClass()).iterator();
104        keys.append("(");
105        values.append("VALUES(");
106        while (iterator.hasNext()) {
107            PropertyDescriptor propertyDescriptor = iterator.next();
108            String name = propertyDescriptor.getName();
109            String dbKey = CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_UNDERSCORE, name);
110            Object o = null;
111            try {
112                o = propertyDescriptor.getReadMethod().invoke(obj);
113            } catch (IllegalAccessException | InvocationTargetException e) {
114                e.printStackTrace();
115            }
116            if (!dbKey.equals("id") || includeId)
117            {
118                keys.append(dbKey);
119                values.append(o);
120                if (iterator.hasNext()) {
121                    keys.append(", ");
122                    values.append(", ");
123                }
124            }
125        }
126        keys.append(")");
127        values.append(")");
128        return keys.append(" ").append(values).toString();
129    }
130
131    public static <T extends DBObject> String getKeyValuePairs(T obj, boolean includeId)
132    {
133        StringBuilder sb = new StringBuilder();
134        Iterator<PropertyDescriptor> iterator = BeanUtils.getBeanProperties(obj.getClass()).iterator();
135        while (iterator.hasNext()) {
136            PropertyDescriptor propertyDescriptor = iterator.next();
137            String name = propertyDescriptor.getName();
138            String dbKey = CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_UNDERSCORE, name);
139            Object o = null;
140            try {
141                o = propertyDescriptor.getReadMethod().invoke(obj);
142            } catch (IllegalAccessException | InvocationTargetException e) {
143                e.printStackTrace();
144            }
145            if (!dbKey.equals("id") || includeId)
146            {
147                sb.append(dbKey);
148                sb.append(" = ");
149                sb.append('\'').append(o).append('\'');
150                if (iterator.hasNext()) {
151                    sb.append(", ");
152                }
153            }
154        }
155        return sb.toString();
156    }
157
158    public static String getKeyValuePair(PropertyDescriptor propertyDescriptor, Object newValue)
159    {
160        StringBuilder sb = new StringBuilder();
161        String name = propertyDescriptor.getName();
162        String dbKey = CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_UNDERSCORE, name);
163        sb.append(dbKey);
164        sb.append(" = ");
165        sb.append('\'').append(newValue).append('\'');
166        return sb.toString();
167    }

169    public static String getTableName(Class<? extends DBObject> cls)
170    {
171        return CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_UNDERSCORE, cls.getSimpleName());
172    }
173
174    /**
175     * Get corresponding SQLite type from the Java type, with special prefix activated by the name<br>
176     * For example: <br>
177     * <pre>
178     * id INTEGER PRIMARY KEY AUTOINCREMENT
179     * </pre>
180     * @param name
181     * @param cls
182     * @return SQLite Type in UPPER CASE
183     */
184    public static String getType(String name, Class<?> cls)
185    {
186        String res = name;
187        String append;
188        switch (cls.getSimpleName())
189        {
190            case "Integer":
191            case "LocalDate":
192            case "LocalDateTime":
193                append = "INTEGER"; break;
194            case "String":
195            case "List":
196                append = "TEXT"; break;
197            default:
198                append = "BLOB"; break;
199        }
200        res += " ";
201        res += append;
202        if ("id".equals(name)) {
203            res += " PRIMARY KEY AUTOINCREMENT";
204        }
205        return res;
206    }
