001package edu.neu.csye6200.helper;
002
003import com.google.common.base.CaseFormat;
004import edu.neu.csye6200.model.DBObject;
005
006import java.beans.PropertyDescriptor;
007import java.lang.reflect.InvocationTargetException;
008import java.util.Iterator;
009
010public class SQLUtils
011{
012    public static String getProperties(Class<? extends DBObject> cls)
013    {
014        StringBuilder sb = new StringBuilder();
015        Iterator<PropertyDescriptor> iterator = BeanUtils.getBeanProperties(cls).iterator();
016        while (iterator.hasNext()) {
017            PropertyDescriptor propertyDescriptor = iterator.next();
018            String name = propertyDescriptor.getName();
019            String dbKey = CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_UNDERSCORE, name);
020            Class<?> returnType = propertyDescriptor.getReadMethod().getReturnType();
021            String sql = getType(dbKey, returnType);
022            if (!sql.isEmpty())
023            {
024                sb.append(sql);
025                if (iterator.hasNext()) {
026                    sb.append(", ");
027                }
028            }
029        }
030        return sb.toString();
031    }
032
033    public static <T extends DBObject> String getKeysAndValues(T obj, boolean includeId)
034    {
035        StringBuilder keys = new StringBuilder();
036        StringBuilder values = new StringBuilder();
037        Iterator<PropertyDescriptor> iterator = BeanUtils.getBeanProperties(obj.getClass()).iterator();
038        keys.append("(");
039        values.append("VALUES(");
040        while (iterator.hasNext()) {
041            PropertyDescriptor propertyDescriptor = iterator.next();
042            String name = propertyDescriptor.getName();
043            String dbKey = CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_UNDERSCORE, name);
044            Object o = null;
045            try {
046                o = propertyDescriptor.getReadMethod().invoke(obj);
047            } catch (IllegalAccessException | InvocationTargetException e) {
048                e.printStackTrace();
049            }
050            if (!dbKey.equals("id") || includeId)
051            {
052                keys.append(dbKey);
053                values.append(o);
054                if (iterator.hasNext()) {
055                    keys.append(", ");
056                    values.append(", ");
057                }
058            }
059        }
060        keys.append(")");
061        values.append(")");
062        return keys.append(" ").append(values).toString();
063    }
064
065    public static <T extends DBObject> String getKeyValuePairs(T obj, boolean includeId)
066    {
067        StringBuilder sb = new StringBuilder();
068        Iterator<PropertyDescriptor> iterator = BeanUtils.getBeanProperties(obj.getClass()).iterator();
069        while (iterator.hasNext()) {
070            PropertyDescriptor propertyDescriptor = iterator.next();
071            String name = propertyDescriptor.getName();
072            String dbKey = CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_UNDERSCORE, name);
073            Object o = null;
074            try {
075                o = propertyDescriptor.getReadMethod().invoke(obj);
076            } catch (IllegalAccessException | InvocationTargetException e) {
077                e.printStackTrace();
078            }
079            if (!dbKey.equals("id") || includeId)
080            {
081                sb.append(dbKey);
082                sb.append(" = ");
083                sb.append('\'').append(o).append('\'');
084                if (iterator.hasNext()) {
085                    sb.append(", ");
086                }
087            }
088        }
089        return sb.toString();
090    }
091
092    public static String getKeyValuePair(PropertyDescriptor propertyDescriptor, Object newValue)
093    {
094        StringBuilder sb = new StringBuilder();
095        String name = propertyDescriptor.getName();
096        String dbKey = CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_UNDERSCORE, name);
097        sb.append(dbKey);
098        sb.append(" = ");
099        sb.append('\'').append(newValue).append('\'');
100        return sb.toString();
101    }
102
103    public static String getTableName(Class<? extends DBObject> cls)
104    {
105        return CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_UNDERSCORE, cls.getSimpleName());
106    }
107
108    /**
109     * Get corresponding SQLite type from the Java type, with special prefix activated by the name<br>
110     * For example: <br>
111     * <pre>
112     * id INTEGER PRIMARY KEY AUTOINCREMENT
113     * </pre>
114     * @param name
115     * @param cls
116     * @return SQLite Type in UPPER CASE
117     */
118    public static String getType(String name, Class<?> cls)
119    {
120        String res = name;
121        String append;
122        switch (cls.getSimpleName())
123        {
124            case "Integer":
125            case "LocalDate":
126            case "LocalDateTime":
127                append = "INTEGER"; break;
128            case "String":
129                append = "TEXT"; break;
130            default:
131                append = "BLOB"; break;
132        }
133        res += " ";
134        res += append;
135        if ("id".equals(name)) {
136            res += " PRIMARY KEY AUTOINCREMENT";
137        }
138        return res;
139    }
