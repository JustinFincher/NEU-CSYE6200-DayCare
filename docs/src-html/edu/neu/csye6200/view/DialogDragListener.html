001package edu.neu.csye6200.view;
002
003import javax.swing.*;
004import java.awt.*;
005import java.awt.event.MouseAdapter;
006import java.awt.event.MouseEvent;
007
008public class DialogDragListener extends MouseAdapter {
009
010    private final JDialog dialog;
011    private Point mouseDownCompCoords = null;
012
013    public DialogDragListener(JDialog dialog) {
014        this.dialog = dialog;
015    }
016
017    public void mouseReleased(MouseEvent e) {
018        mouseDownCompCoords = null;
019    }
020
021    public void mousePressed(MouseEvent e) {
022        mouseDownCompCoords = e.getPoint();
023    }
024
025    public void mouseDragged(MouseEvent e) {
026        Point currCoords = e.getLocationOnScreen();
027        dialog.setLocation(currCoords.x - mouseDownCompCoords.x, currCoords.y - mouseDownCompCoords.y);
028    }
029}
