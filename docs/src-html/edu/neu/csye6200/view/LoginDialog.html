043    private void updateButtonActiveState()
044    {
045        loginButton.setEnabled(isPasswordPresent() && isUserNamePresent());
046    }
047
048    public boolean isUserNamePresent()
049    {
050        String userName = userNameTextField.getText();
051        return userName != null && !userName.isEmpty();
052    }
053
054    public boolean isPasswordPresent()
055    {
056        char[] password = passwordTextField.getPassword();
057        return password.length > 0;
058    }
