001package edu.neu.csye6200.controller;
002
003import edu.neu.csye6200.manager.AdminManager;
004import edu.neu.csye6200.manager.DatabaseManager;
005import edu.neu.csye6200.manager.GUIManager;
006import edu.neu.csye6200.model.Teacher;
007import edu.neu.csye6200.model.TeacherDao;
008import edu.neu.csye6200.view.LoginDialog;
009
010import javax.swing.*;
011import java.util.Optional;
012
013public class LoginDialogController {
014    public LoginDialog getDialog() { return dialog; }
015    private LoginDialog dialog;
016
017    public LoginDialogController(JFrame frame)
018    {
019        dialog = new LoginDialog(frame, true);
020        dialog.loginButton.addActionListener(e -> {
021            login();
022        });
023        dialog.passwordTextField.addActionListener(e -> {
024            login();
025        });
026    }
027
028    public void show()
029    {
030        dialog.dispose();
031        dialog.setUndecorated(true);
032        dialog.setLocationRelativeTo(null);
033        dialog.setVisible(true);
034    }
035
036    private void login()
037    {
038        String userName = dialog.userNameTextField.getText();
039        String password = new String(dialog.passwordTextField.getPassword());
040        if (dialog.isPasswordPresent() && dialog.isUserNamePresent())
041        {
042            Optional<Teacher> teacher = DatabaseManager.getDB().onDemand(TeacherDao.class).findByUserName(userName);
043            if (teacher.isPresent() && teacher.get().isPasswordRight(password))
044            {
045                AdminManager.getInstance().setLoggedInTeacher(teacher.get());
046                dialog.dispose();
047                GUIManager.getInstance().show();
048            }
049        }
050    }
051}
