014public class AlertPanelController {
015
016    public AlertPanelController() {
017        this.panel = new AlertPanel();
018        this.panel.refreshTableButton.addActionListener(e -> {
019            refresh();
020        });
021        refresh();
022    }
023
024    private void refresh() {
025        // mock data
026        panel.scrollContentView.removeAll();
027        DatabaseManager.getDB().onDemand(StudentDao.class).list(Student.class)
028                .stream().limit(10).forEach(s ->
029        {
030            JCheckBox checkBox = new JCheckBox();
031            checkBox.setText(s.getRealName() + " need vaccine on " + s.getWalkInTime().plusYears(1));
032            checkBox.addItemListener(e -> {
033                if(e.getStateChange() == ItemEvent.SELECTED) {
034//                    int i = Arrays.asList(panel.scrollContentView.getComponents()).indexOf(e.getSource());
035//                    list.remove(i);
036                    panel.scrollContentView.remove(checkBox);
037                    panel.scrollContentView.invalidate();
038                    panel.scrollContentView.repaint();
039//                    refresh();
040                }
041            });
042            panel.scrollContentView.add(checkBox);
043        });
044        panel.scrollContentView.invalidate();
045        panel.scrollContentView.repaint();
046    }
047
048    public AlertPanel getPanel() { return panel; }
049    private AlertPanel panel;
