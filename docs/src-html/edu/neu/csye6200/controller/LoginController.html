001package edu.neu.csye6200.controller;
002
003import edu.neu.csye6200.manager.DatabaseManager;
004import edu.neu.csye6200.model.*;
005import edu.neu.csye6200.view.LoginPanel;
006
007import java.awt.*;
008
009
010public class LoginController
011{
012    public LoginPanel getPanel() { return panel; }
013    private LoginPanel panel;
014
015    public LoginController() {
016
017        panel.Login.addActionListener(e->{
018            if(!DatabaseManager.getDB().onDemand(User.class).getUserName().equalsIgnoreCase(panel.Id.getText())){
019                panel.Warnning.setText("*Incorrect Id or Password");
020                panel.Warnning.setForeground(Color.RED);
021                panel.Wrong_id.setText("*");
022                panel.Wrong_id.setForeground(Color.RED);
023            }
024            else if(!DatabaseManager.getDB().onDemand(User.class).getHashedPassword().equalsIgnoreCase(panel.Password.getText())){
025                panel.Warnning.setText("*Incorrect Id or Password");
026                panel.Warnning.setForeground(Color.RED);
027                panel.Wrong_password.setText("*");
028                panel.Wrong_password.setForeground(Color.RED);
029            }
030            else if(!DatabaseManager.getDB().onDemand(User.class).getUserName().equalsIgnoreCase(panel.Id.getText())||!DatabaseManager.getDB().onDemand(User.class).getHashedPassword().equalsIgnoreCase(panel.Password.getText())){
031                panel.Warnning.setText("*Incorrect Id or Password");
032                panel.Warnning.setForeground(Color.RED);
033                panel.Wrong_password.setText("*");
034                panel.Wrong_password.setForeground(Color.RED);
035                panel.Wrong_id.setText("*");
036                panel.Wrong_id.setForeground(Color.RED);
037            }
038            else{
039                panel.setVisible(false);
040            }
041        });
042
043
044    }
045}
