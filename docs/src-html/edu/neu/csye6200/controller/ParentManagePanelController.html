001package edu.neu.csye6200.controller;
002
003import edu.neu.csye6200.helper.FileUtils;
004import edu.neu.csye6200.helper.Log;
005import edu.neu.csye6200.model.DatabaseTableModel;
006import edu.neu.csye6200.model.Parent;
007import edu.neu.csye6200.model.ParentDao;
008import edu.neu.csye6200.view.ParentManagePanel;
009import org.jdesktop.swingx.search.PatternMatcher;
010
011import javax.swing.*;
012import javax.swing.table.TableRowSorter;
013import java.awt.event.MouseAdapter;
014import java.awt.event.MouseEvent;
015import java.util.regex.Pattern;
016
017public class ParentManagePanelController // CONTROLLER
018{
019    private final DatabaseTableModel<Parent, ParentDao> tableModel = new DatabaseTableModel<>(Parent.class, ParentDao.class); // MODEL
020    private TableRowSorter<DatabaseTableModel<Parent, ParentDao>> tableRowSorter = new TableRowSorter<>(tableModel);
021    private final JPopupMenu tableRightClickMenu = new JPopupMenu();
022
023    public ParentManagePanel getPanel() {
024        return panel;
025    }
026    private ParentManagePanel panel; // VIEW
027    
028    public ParentManagePanelController() {
029        panel = new ParentManagePanel();
030        JMenuItem deleteItem = new JMenuItem("Delete");
031        deleteItem.addActionListener(e -> {
032            tableModel.delete(panel.table, panel.table.getSelectedRows());
033        });
034        panel.table.addMouseListener(new MouseAdapter() {
035            @Override
036            public void mousePressed(MouseEvent e) {
037                super.mousePressed(e);
038                maybeShowPopup(e);
039            }
040
041            @Override
042            public void mouseReleased(MouseEvent e) {
043                super.mouseReleased(e);
044                maybeShowPopup(e);
045            }
046
047            private void maybeShowPopup(MouseEvent e) {
048                if (e.isPopupTrigger() && panel.table == e.getSource()) {
049                    int row = panel.table.rowAtPoint(e.getPoint());
050                    int column = panel.table.columnAtPoint(e.getPoint());
051                    tableRightClickMenu.removeAll();
052                    if (row >= 0 && row < panel.table.getRowCount()) {
053                        tableRightClickMenu.add(deleteItem);
054                        if (!panel.table.isRowSelected(row)) {
055                            panel.table.changeSelection(row, column, false, false);
056                        }
057                    } else {
058                        panel.table.clearSelection();
059                    }
060                    tableRightClickMenu.show(e.getComponent(), e.getX(), e.getY());
061                }
062            }
063        });
064        panel.table.setModel(tableModel);
065        tableModel.refresh();
066
067        panel.exportTableButton.addActionListener(e -> {
068
069        });
070        panel.importTableButton.addActionListener(e -> {
071
072        });
073        panel.addParentButton.addActionListener(e -> {
074            tableModel.addEmpty();
075        });
076        panel.refreshTableButton.addActionListener(e -> {
077            tableModel.refresh();
078        });
079
080        panel.searchPanel.addPatternMatcher(new PatternMatcher() {
081            @Override
082            public Pattern getPattern() {
083                return null;
084            }
085
086            @Override
087            public void setPattern(Pattern pattern) {
088                if (pattern != null)
089                {
090                    Log.i(pattern.pattern());
091                    tableRowSorter.setRowFilter(RowFilter.regexFilter(pattern.pattern()));
092                }else {
093                    tableRowSorter.setRowFilter(null);
094                }
095            }
096        });
097        tableRowSorter = new TableRowSorter<>(tableModel);
098        panel.table.setRowSorter(tableRowSorter);
099
100        panel.exportTableButton.addActionListener(event -> {
101            FileUtils.exportCSV(Parent.class, ParentDao.class);
102        });
103
104        panel.importTableButton.addActionListener(event -> {
105            FileUtils.importCSV(Parent.class, ParentDao.class);
106            tableModel.refresh();
107        });
108    }
109}
