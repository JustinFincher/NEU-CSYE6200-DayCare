001package edu.neu.csye6200;
002
003import edu.neu.csye6200.helper.FileUtils;
004import edu.neu.csye6200.helper.Log;
005import edu.neu.csye6200.manager.DatabaseManager;
006import edu.neu.csye6200.manager.GUIManager;
007import org.apache.commons.cli.*;
008
009import javax.swing.*;
010import java.nio.file.Paths;
011
012public class Application {
013
014        public static String name = "DayCare";
015
016        /**
017         * Entry Point
018         *
019         * Arguments include:
020         * -db filePath : specify an existing db file
021         * -verbose : enable verbose logging
022         * -help : print help manual
023         *
024         * @author Haotian Zheng
025         * @param args commandline args
026         */
027        public static void main(String[] args)
028        {
029                System.setProperty("log4j.saveDirectory", FileUtils.getPath());
030                Options options = new Options();
031                options.addOption(Option.builder("db").argName( "pathFile" ).hasArg().desc("The alternative path of the db file").build());
032                options.addOption(Option.builder("verbose").desc("Enable extensive logging").build());
033                options.addOption(Option.builder("help").desc("Print help").build());
034                CommandLineParser parser = new DefaultParser();
035                try {
036                        CommandLine cmd = parser.parse( options, args);
037                        if (cmd.hasOption("help"))
038                        {
039                                HelpFormatter formatter = new HelpFormatter();
040                                formatter.printHelp( name, options );
041                                return;
042                        }
043                        Log.verbose = cmd.hasOption("verbose");
044                        DatabaseManager.getInstance().use(cmd.hasOption("db") ? cmd.getOptionValue("db") : Paths.get(FileUtils.getPath(),"prod.sqlite3").toAbsolutePath().toString());
045                        GUIManager.getInstance().show();
046                } catch (ParseException e) {
047                        e.printStackTrace();
048                }
049        }
050}
