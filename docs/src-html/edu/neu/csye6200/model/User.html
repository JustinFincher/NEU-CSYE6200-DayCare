001package edu.neu.csye6200.model;
002
003import com.google.common.hash.Hashing;
004
005import java.nio.charset.Charset;
006
007public class User extends Person
008{
009    public String getUserName() {
010        return userName;
011    }
012    public void setUserName(String userName) {
013        this.userName = userName;
014    }
015    private String userName;
016
017    public Integer getUserLevel() {
018        return userLevel;
019    }
020    public void setUserLevel(Integer userLevel) {
021        this.userLevel = userLevel;
022    }
023    private Integer userLevel;
024
025    public String getEmailAddress() { return emailAddress; }
026    public void setEmailAddress(String emailAddress) {
027        this.emailAddress = emailAddress;
028    }
029    private String emailAddress;
030
031    public String getHashedPassword() { return hashedPassword; }
032    public void setHashedPassword(String hashedPassword) { this.hashedPassword = hashedPassword; }
033    public void setPassword(String password) {
034        setHashedPassword(Hashing.sha256().hashString(password, Charset.defaultCharset()).toString());
035    }
036    private String hashedPassword;
037}
