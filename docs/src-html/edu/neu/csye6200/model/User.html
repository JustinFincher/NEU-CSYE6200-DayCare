001package edu.neu.csye6200.model;
002
003import com.google.common.hash.Hashing;
004
005import java.nio.charset.Charset;
006import java.util.Map;
007
008public class User extends Person
009{
010    public String getUserName() { return userName == null ? "" : userName; }
011    public void setUserName(String userName) {
012        this.userName = userName;
013    }
014    private String userName;
015
016    public Integer getUserLevel() { return userLevel == null ? 0 : userLevel; }
017    public void setUserLevel(Integer userLevel) {
018        this.userLevel = userLevel;
019    }
020    private Integer userLevel;
021
022    public String getHashedPassword() { return hashedPassword == null ? Hashing.sha256().hashString("password", Charset.defaultCharset()).toString() : hashedPassword; }
023    public void setHashedPassword(String hashedPassword) { this.hashedPassword = hashedPassword; }
024    public void setPassword(String password) {
025        setHashedPassword(Hashing.sha256().hashString(password, Charset.defaultCharset()).toString());
026    }
027    private String hashedPassword;
028    public boolean isPasswordRight(String password)
029    {
030        return hashedPassword.equals(Hashing.sha256().hashString(password, Charset.defaultCharset()).toString());
031    }
032
033    @Override
034    public void saveCsv(Map<String, String> map) {
035        super.saveCsv(map);
036        map.put("user_name",getUserName());
037        map.put("hashed_password",getHashedPassword());
038    }
039
040    @Override
041    public void loadCsv(Map<String, String> map) {
042        super.loadCsv(map);
043        setUserName(map.getOrDefault("user_name",""));
044        setHashedPassword(map.getOrDefault("hashed_password",null));
045    }
046}
