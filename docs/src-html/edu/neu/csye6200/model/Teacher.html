001package edu.neu.csye6200.model;
002
003import java.util.Map;
004
005public class Teacher extends User
006{
007    public Integer getCredits() {
008        return credits;
009    }
010    public void setCredits(Integer credits) {
011        this.credits = credits;
012    }
013    private Integer credits;
014
015    public String getEmailAddress() { return emailAddress == null ? "i@example.com" : emailAddress; }
016    public void setEmailAddress(String emailAddress) {
017        this.emailAddress = emailAddress;
018    }
019    private String emailAddress;
020
021    @Override
022    public void saveCsv(Map<String, String> map) {
023        super.saveCsv(map);
024        map.put("email_address",getEmailAddress());
025    }
026
027    @Override
028    public void loadCsv(Map<String, String> map) {
029        super.loadCsv(map);
030        setEmailAddress(map.getOrDefault("email_address",""));
031    }
032}
