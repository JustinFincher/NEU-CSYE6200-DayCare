001package edu.neu.csye6200.model;
002
003import org.jdbi.v3.sqlobject.config.RegisterBeanMapper;
004import org.jdbi.v3.sqlobject.customizer.BindBean;
005import org.jdbi.v3.sqlobject.customizer.Define;
006import org.jdbi.v3.sqlobject.statement.SqlQuery;
007import org.jdbi.v3.sqlobject.statement.SqlScript;
008import org.jdbi.v3.sqlobject.statement.SqlUpdate;
009
010import java.util.List;
011
012public interface TeacherDao
013{
014    @SqlUpdate("CREATE TABLE IF NOT EXISTS teacher (id INTEGER PRIMARY KEY AUTOINCREMENT, credits INTEGER, user_name TEXT, real_name TEXT, user_level INTEGER)")
015    void createTable();
016
017    @SqlScript("CREATE TABLE IF NOT EXISTS teacher (<sql>)")
018    void createTable(@Define("sql") String sql);
019
020    @SqlQuery("SELECT * FROM teacher")
021    @RegisterBeanMapper(Teacher.class)
022    List<Teacher> listTeachers();
023
024    @SqlQuery("SELECT * FROM teacher where id = ?")
025    @RegisterBeanMapper(Teacher.class)
026    Teacher findTeacher(Integer id);
027
028    @SqlUpdate("INSERT INTO teacher(id, user_name) VALUES (:id, :userName)")
029    void insertTeacher(@BindBean Teacher teacher);
030
031    @SqlScript("INSERT INTO teacher(<key>) VALUES (<value>)")
032    void insertTeacher(@Define("key") String key, @Define("value") String value);
033
034    @SqlScript("UPDATE teacher SET <keyValuePairs> where id = <id>")
035    void updateTeacher(@Define("keyValuePairs") String keyValuePairs, @Define("id") Integer id);
036}
