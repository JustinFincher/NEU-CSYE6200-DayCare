001package edu.neu.csye6200.model;
002
003import java.util.List;
004
005public class ClassroomGroup {
006
007    public void setTeacher(Teacher teacher) {
008        this.teacher = teacher;
009    }
010
011    public Teacher getTeacher() {
012        return teacher;
013    }
014
015    private Teacher teacher;
016    private List<Student> students;
017    private RatioRule rule;
018
019    public boolean canAddStudent(Student student)
020    {
021        return students.size() < rule.getMaxStudentPerTeacher() &&
022                student.getAge() < rule.getMaxAge() &&
023                rule.getMinAge() < student.getAge();
024    }
025
026    public void addStudent(Student student)
027    {
028        students.add(student);
029    }
030}
