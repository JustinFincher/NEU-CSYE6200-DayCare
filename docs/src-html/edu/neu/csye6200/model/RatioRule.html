001package edu.neu.csye6200.model;
002
003import java.util.Map;
004
005public class RatioRule extends DBObject
006{
007    public Integer getMaxGroupsPerRoom() {
008        return maxGroupsPerRoom;
009    }
010    public void setMaxGroupsPerRoom(Integer maxGroupsPerRoom) {
011        this.maxGroupsPerRoom = maxGroupsPerRoom;
012    }
013    public Integer getMaxStudentPerTeacher() {
014        return maxStudentPerTeacher;
015    }
016    public void setMaxStudentPerTeacher(Integer maxStudentPerTeacher) { this.maxStudentPerTeacher = maxStudentPerTeacher; }
017
018    public Integer getMinAgeInMonth() {
019        return minAgeInMonth;
020    }
021    public void setMinAgeInMonth(Integer minAgeInMonth) {
022        this.minAgeInMonth = minAgeInMonth;
023    }
024    public Integer getMaxAgeInMonth() {
025        return maxAgeInMonth;
026    }
027    public void setMaxAgeInMonth(Integer maxAgeInMonth) {
028        this.maxAgeInMonth = maxAgeInMonth;
029    }
030
031    private Integer maxGroupsPerRoom;
032    private Integer maxStudentPerTeacher;
033    private Integer minAgeInMonth;
034    private Integer maxAgeInMonth;
035
036    @Override
037    public void saveCsv(Map<String, String> map) {
038        super.saveCsv(map);
039        map.put("max_groups_per_room",getMaxGroupsPerRoom().toString());
040        map.put("max_student_per_teacher",getMaxStudentPerTeacher().toString());
041        map.put("min_age_in_month",getMinAgeInMonth().toString());
042        map.put("max_age_in_month",getMaxAgeInMonth().toString());
043    }
044
045    @Override
046    public void loadCsv(Map<String, String> map) {
047        super.loadCsv(map);
048        setMaxGroupsPerRoom(Integer.parseInt(map.get("max_groups_per_room")));
049        setMaxStudentPerTeacher(Integer.parseInt(map.get("max_student_per_teacher")));
050        setMinAgeInMonth(Integer.parseInt(map.get("min_age_in_month")));
051        setMaxAgeInMonth(Integer.parseInt(map.get("max_age_in_month")));
052    }
053}
