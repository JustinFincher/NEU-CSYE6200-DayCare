001package edu.neu.csye6200.model;
002
003import java.time.Instant;
004import java.time.LocalDateTime;
005import java.time.ZoneId;
006import java.time.ZoneOffset;
007import java.util.List;
008import java.util.Map;
009
010/**
011 * Student Data Model
012 * Should has these fields:
013 *  - grade
014 *  - walk-in date
015 *  - immunization record
016 *
017 * @see <a href="https://www.notion.so/haotianzheng/Student-256835304927473cb082a6feabdb5d8e">Student Data On Notion</a>
018 * @author Haotian Zheng
019 */
020public class Student extends User
021{
022    public LocalDateTime getWalkInTime() { return walkInTime == null ? LocalDateTime.now() : walkInTime; }
023    public void setWalkInTime(LocalDateTime walkInTime) { this.walkInTime = walkInTime; }
024    private LocalDateTime walkInTime;
025
026    public Integer getGrade() { return grade == null ? 100 : grade; }
027    public void setGrade(Integer grade) { this.grade = grade; }
028    private Integer grade;
029
030    @Override
031    public void loadCsv(Map<String, String> map) {
032        super.loadCsv(map);
033        setGrade(Integer.parseInt(map.getOrDefault("grade", "0")));
034        setWalkInTime(Instant.ofEpochSecond(Long.parseLong(map.getOrDefault("walk_in_time","0"))).atZone(ZoneId.systemDefault()).toLocalDateTime());
035    }
036
037    @Override
038    public void saveCsv(Map<String, String> map) {
039        super.saveCsv(map);
040        map.put("grade",getGrade().toString());
041        map.put("walk_in_time", String.valueOf(getWalkInTime().toEpochSecond(ZoneOffset.UTC)));
042    }
043}
