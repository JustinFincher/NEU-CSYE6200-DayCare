001package edu.neu.csye6200.model;
002
003import edu.neu.csye6200.helper.Log;
004import edu.neu.csye6200.helper.SQLUtils;
005import edu.neu.csye6200.helper.annotation.SQLLoggingFactory;
006import org.jdbi.v3.core.result.ResultBearing;
007import org.jdbi.v3.sqlobject.SqlObject;
008import org.jdbi.v3.sqlobject.customizer.Define;
009import org.jdbi.v3.sqlobject.statement.GetGeneratedKeys;
010import org.jdbi.v3.sqlobject.statement.SqlQuery;
011import org.jdbi.v3.sqlobject.statement.SqlUpdate;
012
013import java.util.List;
014import java.util.Map;
015import java.util.Optional;
016import java.util.function.BiConsumer;
017
018/**
019 *
020 * CRUD DAO, a generic DAO class for all {@link DBObject DBObject} subclass <br>
021 * How to use:<br>
022 * <pre>
023 * {@literal
024 * // Let's say you have a subclass of CrudDao called PeopleDao:
025 * (remember to add @RegisterBeanMapper(People.class) annotation on your Dao)
026 * public interface PeopleDao extends CrudDao<People>
027 * {
028 * }
029 *
030 * // then:
031 * PeopleDao dao = DatabaseManager.getDB().onDemand(PeopleDao.class);
032 * // then you can call all base methods declared in <CrudDao>, like:
033 * dao.createTable("people", "id INTEGER PRIMARY KEY");
034 * // but most of the time you don't need to hard code SQL queries, as you can use <SQLUtils>:
035 * dao.createTable(SQLUtils.getTableName(People.class), SQLUtils.getProperties(People.class));
036 * }
037 * </pre>
038 *
039 * @author Haotian Zheng
040 * @see DBObject
041 * @see DatabaseTableModel
042 * @param <T> A placeholder for the data model class later to be specified
043 */
044public interface CrudDao<T extends DBObject> extends SqlObject
045{
046    /**
047     * Create a table in database if it does not exist
048     * @param tableName table name, usually acquired via SQLUtils.getTableName(YourDataModel.class)
049     * @param sql sql query for table properties, also with a helper method SQLUtils.getProperties(YourDataModel.class)
050     */
051    @SqlUpdate("CREATE TABLE IF NOT EXISTS <tableName> (<sql>)")
052    void createTable(@Define("tableName") String tableName, @Define("sql") String sql);
053    default void createTable(Class<T> className)
054    {
055        getHandle().createUpdate("CREATE TABLE IF NOT EXISTS <tableName> (<sql>)")
056                .define("tableName", SQLUtils.getTableName(className))
057                .define("sql", SQLUtils.getAllKeysInString(className))
058                .execute();
059    }
060
061    /**
062     * Insert an entry into the database
063     * @param tableName table name, usually acquired via SQLUtils.getTableName(YourDataModel.class)
064     * @param keysAndValues usually acquired via SQLUtils.getKeysAndValues(model, false)
065     */
066    @GetGeneratedKeys
067    @SqlUpdate("INSERT INTO <tableName> <keysAndValues>")
068    int insert(@Define("tableName") String tableName, @Define("keysAndValues") String keysAndValues);
069
070    default void insert(T object, Class<T> className)
071    {
072        String tableName = SQLUtils.getTableName(className);
073        String bindings = SQLUtils.getKeysAndValueBindingsInCreateString(object, false);
074        Map<String, ?> map = SQLUtils.getKeysAndValuesInMap(object);
075        try {
076            getHandle().createUpdate("INSERT INTO <tableName> " + bindings)
077                    .define("tableName", tableName)
078                    .bindMap(map)
079                    .execute();
080        }catch (Exception e)
081        {
082            Log.e(e.getMessage());
083            e.printStackTrace();
084        }
085    }
086
087    @SqlQuery("SELECT * FROM <tableName> where id = ?")
088    Optional<T> findById(@Define("tableName") String tableName, Integer id);
089    default T find(Class<T> className, Integer id)
090    {
091        String tableName = SQLUtils.getTableName(className);
092        return getHandle().createQuery("SELECT * FROM <tableName> where id = :id")
093                .define("tableName", tableName)
094                .bind("id", id)
095                .mapTo(className)
096                .one();
097    }
098
099    @SqlQuery("SELECT * FROM <tableName>")
100    List<T> list(@Define("tableName") String tableName);
101
102    default List<T> list(Class<T> className)
103    {
104        String tableName = SQLUtils.getTableName(className);
105        return getHandle().createQuery("SELECT * FROM <tableName>")
106                .define("tableName", tableName)
107                .mapTo(className)
108                .list();
109    }
110
111    default boolean isTableEmpty(Class<T> className)
112    {
113        return list(className).isEmpty();
114    }
115
116    @SqlUpdate("UPDATE <tableName> SET <keyValuePairs> WHERE id = <id>")
117    void update(@Define("tableName") String tableName, @Define("keyValuePairs") String keyValuePairs, @Define("id") Integer id);
118
119    default void update(T object, Class<T> className)
120    {
121        String tableName = SQLUtils.getTableName(className);
122        String bindings = SQLUtils.getKeysAndValueBindingsInUpdateString(object, false);
123        Map<String, ?> map = SQLUtils.getKeysAndValuesInMap(object);
124        String query = "UPDATE <tableName> SET " + bindings + " WHERE id = :id";
125        getHandle().createUpdate(query)
126                .define("tableName", tableName)
127                .bindMap(map)
128                .bind("id", object.getId())
129                .execute();
130    }
131
132    @SqlUpdate("DELETE FROM <tableName> WHERE id = <id>")
133    void deleteById(@Define("tableName") String tableName, @Define("id") Integer id);
134}
