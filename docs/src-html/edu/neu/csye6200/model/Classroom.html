001package edu.neu.csye6200.model;
002
003import edu.neu.csye6200.helper.annotation.JavaBeansIgnore;
004import edu.neu.csye6200.manager.DatabaseManager;
005
006import java.util.Arrays;
007import java.util.List;
008import java.util.Map;
009import java.util.Objects;
010import java.util.stream.Collectors;
011
012public class Classroom extends DBObject
013{
014    @JavaBeansIgnore
015    public List<ClassroomGroup> getGroups()
016    {
017        return Arrays.stream(groupIds.split(","))
018                .map(Integer::parseInt)
019                .map(i -> DatabaseManager.getDB()
020                        .onDemand(ClassroomGroupDao.class)
021                        .find(ClassroomGroup.class, i))
022                .filter(Objects::nonNull)
023                .collect(Collectors.toList());
024    }
025    @JavaBeansIgnore
026    public void setGroups(List<ClassroomGroup> groups)
027    {
028        groupIds = groups.stream().map(DBObject::getId).map(Object::toString).collect(Collectors.joining(","));
029    }
030    public String getGroupIds() { return groupIds; }
031    public void setGroupIds(String groupIds) { this.groupIds = groupIds; }
032    private String groupIds;
033
034    public String getRoomNumber() { return roomNumber; }
035    public void setRoomNumber(String roomNumber) { this.roomNumber = roomNumber; }
036    private String roomNumber;
037
038    @Override
039    public void saveCsv(Map<String, String> map) {
040        super.saveCsv(map);
041        map.put("room_number", getRoomNumber());
042    }
043
044    @Override
045    public void loadCsv(Map<String, String> map) {
046        super.loadCsv(map);
047        setRoomNumber(map.getOrDefault("room_number",""));
048    }
049}
