011
012    Classroom()
013    {
014        // DB
015        List<RatioRule> rules = new ArrayList<>();
016
017        rules.forEach(ratioRule -> {
018            Integer maxGroupsPerRoom = ratioRule.getMaxGroupsPerRoom();
019            // groups.add()
020            for (int i = 0; i < maxGroupsPerRoom; i++) {
021                groups.add(new ClassroomGroup());
022            }
023
024        });
025    }
026
027    public void addTeacher(Teacher teacher)
028    {
029        groups.forEach(classroomGroup -> {
030            while (classroomGroup.getTeacher()==null){
031                classroomGroup.setTeacher(teacher);
032            }
033        });
034    }
