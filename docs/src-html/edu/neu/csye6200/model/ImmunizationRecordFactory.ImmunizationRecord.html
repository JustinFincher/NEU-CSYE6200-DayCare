001package edu.neu.csye6200.model;
002
003import java.util.HashMap;
004import java.util.List;
005import java.util.Map;
006
007/**
008 *
009 * @author Dawei
010 */
011public class ImmunizationRecordFactory extends DBObject {
012    public class ImmunizationRecord extends DBObject{
013        private Map<Vaccine, List<Dose>> record;
014        private boolean isComplete;
015        
016        public ImmunizationRecord() {
017            record = new HashMap<>();
018            isComplete = false;
019        }
020        
021        public void addNewVaccine(Vaccine v, List<Dose> doses) {
022            record.put(v, doses);
023        }
024        
025        public void addNewDose(Vaccine v, Dose d) {
026            record.get(v).add(d);
027        }
028        
029        public boolean isComplete() {
030            for (List<Dose> doses : record.values()) {
031                for (Dose d : doses) {
032                    if (!d.getCheck()) {
033                        return false;
034                    }
035                }
036            }
037            return true;
038        }
039    }
040    
041    public ImmunizationRecord getImmunizationRecord() {
042        return new ImmunizationRecord();
043    }
044}
