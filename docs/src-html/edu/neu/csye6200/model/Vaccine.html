008public class Vaccine extends DBObject {
009    private String name;
010    private String description;
011    
012    public Vaccine(String name) {
013        this(name, "No Description.");
014    }
015    
016    public Vaccine(String name, String description) {
017        this.name = name;
018        this.description = description;
019    }
020    
021    public String getName() {
022        return this.name;
023    }
024    
025    public void setName(String name) {
026        this.name = name;
027    }
028    
029    public String getDescription() {
030        return this.description;
031    }
032    
033    public void setDescription(String description) {
034        this.description = description;
035    }
036}
